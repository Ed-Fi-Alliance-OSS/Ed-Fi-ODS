HQL:
select 
	comp_aaa.Id As __BaseEntityId__,
	comp_aaa.EducationOrganizationId as PK1a_EducationOrganizationId,
	comp_aaa.StudentUSI as PK1a_StudentUSI
from 
	EdFi.Ods.Entities.NHibernate.QueryModels.StudentEducationOrganizationAssociationAggregate.EdFi.StudentEducationOrganizationAssociationQ comp_aaa
where comp_aaa.Id = :Id AND (comp_aaa.EducationOrganizationId IN (
                        SELECT fltr_aan.EducationOrganizationId 
                        FROM EdFi.Ods.Entities.NHibernate.QueryModels.Views.auth_EducationOrganizationIdToLocalEducationAgencyId fltr_aan 
                        WHERE fltr_aan.LocalEducationAgencyId IN (:LocalEducationAgencyId))) AND (comp_aaa.StudentUSI IN (
                            SELECT fltr_aab.StudentUSI 
                            FROM EdFi.Ods.Entities.NHibernate.QueryModels.Views.auth_LocalEducationAgencyIdToStudentUSI fltr_aab 
                            WHERE fltr_aab.LocalEducationAgencyId IN (:LocalEducationAgencyId)))
order by comp_aaa.EducationOrganizationId,
	comp_aaa.StudentUSI
